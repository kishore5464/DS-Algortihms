#include <iostream>
using namespace std;
void permutations(char *input,int i){
    //Base case
    if(input[i]=='\0'){
        cout<<input<<",";
        return;
    }
    
    //recursive case
    for(int j=i;input[j]!='\0';j++){
        swap(input[i],input[j]);
        permutations(input,i+1);
        swap(input[i],input[j]);
    }
}

int main() {
    char input[100];
    cin>>input;
    
    permutations(input,0);
}
